584  template <typename T>
585  Vector<T, 2> MonotonicCatmullRom(
586  const Vector<T, 2>& v0,
587  const Vector<T, 2>& v1,
588  const Vector<T, 2>& v2,
589  const Vector<T, 2>& v3,
590  T f)
591  {
592  static const T two = static_cast<T>(2);
593  static const T three = static_cast<T>(3);
594 
595  Vector<T, 2> d1 = (v2 - v0) / two;
596  Vector<T, 2> d2 = (v3 - v1) / two;
597  Vector<T, 2> D1 = v2 - v1;
598 
599  if (std::fabs(D1.x) < std::numeric_limits<T>::epsilon() ||
600  Sign(D1.x) != Sign(d1.x) ||
601  Sign(D1.x) != Sign(d2.x))
602  {
603  d1.x = d2.x = 0;
604  }
605 
606  if (std::fabs(D1.y) < std::numeric_limits<T>::epsilon() ||
607  Sign(D1.y) != Sign(d1.y) ||
608  Sign(D1.y) != Sign(d2.y))
609  {
610  d1.y = d2.y = 0;
611  }
612 
613  Vector<T, 2> a3 = d1 + d2 - two * D1;
614  Vector<T, 2> a2 = three * D1 - two * d1 - d2;
615  Vector<T, 2> a1 = d1;
616  Vector<T, 2> a0 = v1;
617 
618  return a3 * Cubic(f) + a2 * Square(f) + a1 * f + a0;
619  }
