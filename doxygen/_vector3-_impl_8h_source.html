619  template <typename T>
620  Vector<T, 3> MonotonicCatmullRom(
621  const Vector<T, 3>& v0,
622  const Vector<T, 3>& v1,
623  const Vector<T, 3>& v2,
624  const Vector<T, 3>& v3,
625  T f)
626  {
627  static const T two = static_cast<T>(2);
628  static const T three = static_cast<T>(3);
629 
630  Vector<T, 3> d1 = (v2 - v0) / two;
631  Vector<T, 3> d2 = (v3 - v1) / two;
632  Vector<T, 3> D1 = v2 - v1;
633 
634  if (std::fabs(D1.x) < std::numeric_limits<float>::epsilon() ||
635  Sign(D1.x) != Sign(d1.x) ||
636  Sign(D1.x) != Sign(d2.x))
637  {
638  d1.x = d2.x = 0;
639  }
640 
641  if (std::fabs(D1.y) < std::numeric_limits<float>::epsilon() ||
642  Sign(D1.y) != Sign(d1.y) ||
643  Sign(D1.y) != Sign(d2.y))
644  {
645  d1.y = d2.y = 0;
646  }
647 
648  if (std::fabs(D1.z) < std::numeric_limits<float>::epsilon() ||
649  Sign(D1.z) != Sign(d1.z) ||
650  Sign(D1.z) != Sign(d2.z))
651  {
652  d1.z = d2.z = 0;
653  }
654 
655  Vector<T, 3> a3 = d1 + d2 - two * D1;
656  Vector<T, 3> a2 = three * D1 - two * d1 - d2;
657  Vector<T, 3> a1 = d1;
658  Vector<T, 3> a0 = v1;
659 
660  return a3 * Cubic(f) + a2 * Square(f) + a1 * f + a0;
661  }
