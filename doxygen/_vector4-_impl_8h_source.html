606  template <typename T>
607  Vector<T, 4> MonotonicCatmullRom(
608  const Vector<T, 4>& v0,
609  const Vector<T, 4>& v1,
610  const Vector<T, 4>& v2,
611  const Vector<T, 4>& v3,
612  T f)
613  {
614  static const T two = static_cast<T>(2);
615  static const T three = static_cast<T>(3);
616 
617  Vector<T, 4> d1 = (v2 - v0) / two;
618  Vector<T, 4> d2 = (v3 - v1) / two;
619  Vector<T, 4> D1 = v2 - v1;
620 
621  if (std::fabs(D1.x) < std::numeric_limits<float>::epsilon() ||
622  Sign(D1.x) != Sign(d1.x) ||
623  Sign(D1.x) != Sign(d2.x))
624  {
625  d1.x = d2.x = 0;
626  }
627 
628  if (std::fabs(D1.y) < std::numeric_limits<float>::epsilon() ||
629  Sign(D1.y) != Sign(d1.y) ||
630  Sign(D1.y) != Sign(d2.y))
631  {
632  d1.y = d2.y = 0;
633  }
634 
635  if (std::fabs(D1.z) < std::numeric_limits<float>::epsilon() ||
636  Sign(D1.z) != Sign(d1.z) ||
637  Sign(D1.z) != Sign(d2.z))
638  {
639  d1.z = d2.z = 0;
640  }
641 
642  if (std::fabs(D1.w) < std::numeric_limits<float>::epsilon() ||
643  Sign(D1.w) != Sign(d1.w) ||
644  Sign(D1.w) != Sign(d2.w))
645  {
646  d1.w = d2.w = 0;
647  }
648 
649  Vector<T, 4> a3 = d1 + d2 - two * D1;
650  Vector<T, 4> a2 = three * D1 - two * d1 - d2;
651  Vector<T, 4> a1 = d1;
652  Vector<T, 4> a0 = v1;
653 
654  return a3 * Cubic(f) + a2 * Square(f) + a1 * f + a0;
655  }
